def extract_swift_features(path):
    bild = cv2.imread(path, cv2.IMREAD_GRAYSCALE)
    bild = cv2.resize(bild, (150, 150))
    hog = cv2.HOGDescriptor()
    
    descriptors = hog.compute(bild, winStride=(64, 64))
    
    return descriptors
